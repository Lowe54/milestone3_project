db.createCollection("recipies", {
	validator: {
		$jsonSchema: {
		bsonType: "object",
		required: ["recipie_title", "recipie_instructions", "recipie_ingredients", "recipie_allergins"],
		properties:{
			recipie_title:{
				bsonType:"string",
				description:"The title must be a string and is required"
			},
			recipie_instructions:{
				bsonType:"string",
				description:"The recipie instructions must be a string and is required"
			},
			"recipie_ingredients": {
				bsonType:"string",
				description:"The recipie ingredients must be a string and is required"
			},
			recipie_allergins: {
				enum: [ "Gluten", "Crustacean", "Eggs", "Fish", "Peanuts", "Soybeans", "Milk", "Nuts", "Celery", "Mustard", "Sesame", "Lupin", "Molluscs", null ],
				description:"The recipie allergins can only be one of the enum values and is required"
			}
		}
		}
	}

})
